from sklearn.cluster import KMeans
k_values = [2, 3, 4, 5, 6, 7, 8]
inertias, silhouettes = [], []

for k in k_values:
    km = KMeans(n_clusters=k, random_state=0, n_init=10)
    labels = km.fit_predict(X_scaled)
    inertias.append(km.inertia_)
    silhouettes.append(silhouette_score(X_scaled, labels))

plt.figure()
plt.plot(k_values, inertias, marker='o')
plt.title("Elbow (K-Means)")

plt.plot(k_values, silhouettes, marker='o')
plt.title("Silhouette (K-Means)")

best_k = k_values[int(np.argmax(silhouettes))]
print("Chosen k:", best_k)

#Final K-Means with chosen k and cluster profiles

# target (median-impute Y)
import numpy as np
from sklearn.metrics import mean_squared_error, r2_score
from sklearn.model_selection import KFold, cross_val_score
y = data[target_col]
impY = SimpleImputer(strategy="median")
y_imp = impY.fit_transform(y.to_frame()).ravel()

# training and testing split
X_train, X_test, y_train, y_test = train_test_split(X_imp, y_imp, test_size=0.2, random_state=42)

# Model 1: Random Forest
rf = RandomForestRegressor(n_estimators=350, random_state=42, n_jobs=-1, min_samples_leaf=2)
rf.fit(X_train, y_train)
pred_rf = rf.predict(X_test)
rf_mse  = mean_squared_error(y_test, pred_rf)   # no 'squared' kwarg
rf_rmse = np.sqrt(rf_mse)
rf_r2   = r2_score(y_test, pred_rf)
print(f"Random Forest: RMSE={rf_rmse:.1f}, R^2={rf_r2:.3f}")

rf_imp = pd.Series(rf.feature_importances_, index=feature_cols).sort_values(ascending=False)
top5_rf = rf_imp.head(5)
display(top5_rf)

# Model 2: Lasso (linear)
scaler2 = StandardScaler()
X_train_std = scaler2.fit_transform(X_train)
X_test_std  = scaler2.transform(X_test)

lasso = LassoCV(cv=5, random_state=42, n_jobs=-1, max_iter=10000)
lasso.fit(X_train_std, y_train)
pred_ls = lasso.predict(X_test_std)
ls_mse  = mean_squared_error(y_test, pred_ls)   # no 'squared' kwarg
ls_rmse = np.sqrt(ls_mse)
ls_r2   = r2_score(y_test, pred_ls)
print(f"Lasso (linear): RMSE={ls_rmse:.1f}, R^2={ls_r2:.3f}")

ls_coef = pd.Series(lasso.coef_, index=feature_cols).fillna(0.0)
top5_lasso = ls_coef.abs().sort_values(ascending=False).head(5)
display(top5_lasso)

# 5-fold cross-validation for both models
kf = KFold(n_splits=5, shuffle=True, random_state=42)
rf_cv = cross_val_score(rf, X_imp, y_imp, cv=kf, scoring="r2")
lasso_cv = cross_val_score(lasso, scaler2.fit_transform(X_imp), y_imp, cv=kf, scoring="r2")

print("Random Forest CV R^2 (mean):", rf_cv.mean())
print("Lasso CV R^2 (mean):", lasso_cv.mean())
